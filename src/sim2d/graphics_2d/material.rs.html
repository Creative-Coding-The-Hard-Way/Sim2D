1use {

9/// The shader entrypoint name, always defaults to 'main'.
10const SHADER_ENTRYPOINT: &CStr = c"main";
11
12/// Materials are used to style mesh properties.
13///
14/// Materials are immutable and can be shared by meshes.
15#[derive(Debug)]
16pub struct Material {
17    pipeline: raii::Pipeline,
18}
19
20impl Material {
21    /// Builds a pipeline layout for use with Material pipelines.
22    pub(super) fn create_pipeline_layout(
23        gfx: &Gfx,
24        texture_atlas_descriptor_set_layout: &raii::DescriptorSetLayout,
25        frame_constants_descriptor_set_layout: &raii::DescriptorSetLayout,
26    ) -> Result<raii::PipelineLayout> {
27        let raw_descriptor_set_layouts = [
28            texture_atlas_descriptor_set_layout.raw,
29            frame_constants_descriptor_set_layout.raw,
30        ];
31        let push_constant_ranges = [vk::PushConstantRange {
32            stage_flags: vk::ShaderStageFlags::VERTEX,
33            offset: 0,
34            size: 8 + 8 + 4,
35        }];
36        raii::PipelineLayout::new(
37            "FirstTriangle",
38            gfx.vulkan.device.clone(),
39            &vk::PipelineLayoutCreateInfo {
40                set_layout_count: raw_descriptor_set_layouts.len() as u32,
41                p_set_layouts: raw_descriptor_set_layouts.as_ptr(),
42                push_constant_range_count: push_constant_ranges.len() as u32,
43                p_push_constant_ranges: push_constant_ranges.as_ptr(),
44                ..Default::default()
45            },
46        )
47    }
48
49    /// Creates a new material for use when rendering meshes.
50    pub(super) fn new(
51        gfx: &Gfx,
52        pipeline_layout: &raii::PipelineLayout,
53        vertex_shader_module: &raii::ShaderModule,
54        fragment_shader_module: &raii::ShaderModule,
55    ) -> Result<Self> {
56        let stages = [
57            vk::PipelineShaderStageCreateInfo {
58                stage: vk::ShaderStageFlags::VERTEX,
59                module: vertex_shader_module.raw,
60                p_name: SHADER_ENTRYPOINT.as_ptr(),
61                ..Default::default()
62            },
63            vk::PipelineShaderStageCreateInfo {
64                stage: vk::ShaderStageFlags::FRAGMENT,
65                module: fragment_shader_module.raw,
66                p_name: SHADER_ENTRYPOINT.as_ptr(),
67                ..Default::default()
68            },
69        ];
70
71        let vertex_input_state = vk::PipelineVertexInputStateCreateInfo {
72            vertex_binding_description_count: 0,
73            vertex_attribute_description_count: 0,
74            ..Default::default()
75        };
76        let input_assembly_state = vk::PipelineInputAssemblyStateCreateInfo {
77            topology: vk::PrimitiveTopology::TRIANGLE_LIST,
78            primitive_restart_enable: vk::FALSE,
79            ..Default::default()
80        };
81        let tesselation_state = vk::PipelineTessellationStateCreateInfo {
82            patch_control_points: 0,
83            ..Default::default()
84        };
85        let rasterization_state = vk::PipelineRasterizationStateCreateInfo {
86            depth_clamp_enable: vk::FALSE,
87            rasterizer_discard_enable: vk::FALSE,
88            polygon_mode: vk::PolygonMode::FILL,
89            cull_mode: vk::CullModeFlags::NONE,
90            front_face: vk::FrontFace::COUNTER_CLOCKWISE,
91            depth_bias_enable: vk::FALSE,
92            depth_bias_constant_factor: 0.0,
93            depth_bias_clamp: 0.0,
94            depth_bias_slope_factor: 0.0,
95            line_width: 1.0,
96            ..Default::default()
97        };
98        let multisample_state = vk::PipelineMultisampleStateCreateInfo {
99            rasterization_samples: vk::SampleCountFlags::TYPE_1,
100            sample_shading_enable: vk::FALSE,
101            min_sample_shading: 1.0,
102            p_sample_mask: std::ptr::null(),
103            alpha_to_coverage_enable: vk::FALSE,
104            alpha_to_one_enable: vk::FALSE,
105            ..Default::default()
106        };
107        let depth_stencil_state = vk::PipelineDepthStencilStateCreateInfo {
108            depth_test_enable: vk::FALSE,
109            depth_write_enable: vk::FALSE,
110            depth_compare_op: vk::CompareOp::LESS,
111            depth_bounds_test_enable: vk::FALSE,
112            stencil_test_enable: vk::FALSE,
113            min_depth_bounds: 0.0,
114            max_depth_bounds: 1.0,
115            ..Default::default()
116        };
117        let color_blend_statetachment_state =
118            vk::PipelineColorBlendAttachmentState {
119                blend_enable: vk::TRUE,
120                src_color_blend_factor: vk::BlendFactor::SRC_ALPHA,
121                dst_color_blend_factor: vk::BlendFactor::ONE_MINUS_SRC_ALPHA,
122                color_blend_op: vk::BlendOp::ADD,
123                src_alpha_blend_factor: vk::BlendFactor::ONE,
124                dst_alpha_blend_factor: vk::BlendFactor::ZERO,
125                alpha_blend_op: vk::BlendOp::ADD,
126                color_write_mask: vk::ColorComponentFlags::RGBA,
127            };
128        let color_blend_state = vk::PipelineColorBlendStateCreateInfo {
129            logic_op_enable: vk::FALSE,
130            logic_op: vk::LogicOp::COPY,
131            attachment_count: 1,
132            p_attachments: &color_blend_statetachment_state,
133            blend_constants: [0.0, 0.0, 0.0, 0.0],
134            ..Default::default()
135        };
136
137        let dynamic_states =
138            [vk::DynamicState::VIEWPORT, vk::DynamicState::SCISSOR];
139        let dynamic_state = vk::PipelineDynamicStateCreateInfo {
140            dynamic_state_count: dynamic_states.len() as u32,
141            p_dynamic_states: dynamic_states.as_ptr(),
142            ..Default::default()
143        };
144
145        let viewport_state = vk::PipelineViewportStateCreateInfo {
146            viewport_count: 1,
147            scissor_count: 1,
148            ..Default::default()
149        };
150
151        let color_attachment_formats = [gfx.swapchain.format()];
152        let mut rendering_info = vk::PipelineRenderingCreateInfo {
153            view_mask: 0,
154            color_attachment_count: 1,
155            p_color_attachment_formats: color_attachment_formats.as_ptr(),
156            ..Default::default()
157        };
158
159        let create_info = vk::GraphicsPipelineCreateInfo {
160            stage_count: stages.len() as u32,
161            p_stages: stages.as_ptr(),
162            p_vertex_input_state: &vertex_input_state,
163            p_input_assembly_state: &input_assembly_state,
164            p_tessellation_state: &tesselation_state,
165            p_viewport_state: &viewport_state,
166            p_rasterization_state: &rasterization_state,
167            p_multisample_state: &multisample_state,
168            p_depth_stencil_state: &depth_stencil_state,
169            p_color_blend_state: &color_blend_state,
170            p_dynamic_state: &dynamic_state,
171            layout: pipeline_layout.raw,
172            render_pass: vk::RenderPass::null(),
173            subpass: 0,
174            base_pipeline_handle: vk::Pipeline::null(),
175            base_pipeline_index: 0,
176            ..Default::default()
177        }
178        .push_next(&mut rendering_info);
179
180        let pipeline = raii::Pipeline::new_graphics_pipeline(
181            gfx.vulkan.device.clone(),
182            &create_info,
183        )
184        .context("Unable to create pipeline!")?;
185
186        Ok(Self { pipeline })
187    }
188
189    /// Returns the pipeline used by this material.
190    pub fn pipeline(&self) -> &raii::Pipeline {
191        &self.pipeline
192    }
