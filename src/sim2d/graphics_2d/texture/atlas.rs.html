8/// Used to constrain the variable descriptor array and pool sizes. This is well
9/// below the 500_000 limit imposed by the spec for the minimum supported number
10/// of UPDATE_AFTER_BIND descriptors in a descriptor poolw.
11const MAX_TEXTURES: u32 = 10_000;
12
13/// The TextureAtlas maintains the per-frame descriptor sets used for binding
14/// textures.
15///
16/// New textures can be added to the atlas at any time and will be available for
17/// use in the next frame.
18pub struct TextureAtlas {
19    // All textures currently in-use by the atlas.
20    textures: Vec<Texture>,
21
22    // The sampler used when reading textures.
23    _sampler: raii::Sampler,
24
25    descriptor_set_layout: raii::DescriptorSetLayout,
26    descriptor_set: vk::DescriptorSet,
27    _descriptor_pool: raii::DescriptorPool,
28
29    pipeline_layout: raii::PipelineLayout,
30}
31
32impl TextureAtlas {
33    pub fn new(gfx: &Gfx) -> Result<Self> {
34        let sampler = raii::Sampler::new(
35            "TextureAtlas Immutable Sampler",
36            gfx.vulkan.device.clone(),
37            &vk::SamplerCreateInfo {
38                mag_filter: vk::Filter::LINEAR,
39                min_filter: vk::Filter::NEAREST,
40                mipmap_mode: vk::SamplerMipmapMode::LINEAR,
41                address_mode_u: vk::SamplerAddressMode::CLAMP_TO_EDGE,
42                address_mode_v: vk::SamplerAddressMode::CLAMP_TO_EDGE,
43                address_mode_w: vk::SamplerAddressMode::CLAMP_TO_EDGE,
44                mip_lod_bias: 0.0,
45                anisotropy_enable: vk::FALSE,
46                max_anisotropy: 0.0,
47                compare_enable: vk::FALSE,
48                compare_op: vk::CompareOp::ALWAYS,
49                min_lod: 0.0,
50                max_lod: vk::LOD_CLAMP_NONE,
51                border_color: vk::BorderColor::FLOAT_OPAQUE_WHITE,
52                unnormalized_coordinates: vk::FALSE,
53                ..Default::default()
54            },
55        )
56        .context("Unable to create texture atlas sampler!")?;
57
58        let descriptor_set_layout = {
59            let bindings = [
60                vk::DescriptorSetLayoutBinding {
61                    binding: 0,
62                    descriptor_type: vk::DescriptorType::SAMPLER,
63                    descriptor_count: 1,
64                    stage_flags: vk::ShaderStageFlags::FRAGMENT,
65                    p_immutable_samplers: &sampler.raw,
66                    ..Default::default()
67                },
68                vk::DescriptorSetLayoutBinding {
69                    binding: 1,
70                    descriptor_type: vk::DescriptorType::SAMPLED_IMAGE,
71                    descriptor_count: MAX_TEXTURES,
72                    stage_flags: vk::ShaderStageFlags::FRAGMENT,
73                    p_immutable_samplers: std::ptr::null(),
74                    _marker: std::marker::PhantomData,
75                },
76            ];
77            let binding_flags = [
78                vk::DescriptorBindingFlags::empty(),
79                vk::DescriptorBindingFlags::PARTIALLY_BOUND
80                    | vk::DescriptorBindingFlags::VARIABLE_DESCRIPTOR_COUNT
81                    | vk::DescriptorBindingFlags::UPDATE_AFTER_BIND
82                    | vk::DescriptorBindingFlags::UPDATE_UNUSED_WHILE_PENDING,
83            ];
84            let mut binding_flags_create_info =
85                vk::DescriptorSetLayoutBindingFlagsCreateInfo {
86                    binding_count: binding_flags.len() as u32,
87                    p_binding_flags: binding_flags.as_ptr(),
88                    ..Default::default()
89                };
90            let create_info = vk::DescriptorSetLayoutCreateInfo {
91                flags:
92                    vk::DescriptorSetLayoutCreateFlags::UPDATE_AFTER_BIND_POOL,
93                binding_count: bindings.len() as u32,
94                p_bindings: bindings.as_ptr(),
95                ..Default::default()
96            }
97            .push_next(&mut binding_flags_create_info);
98            raii::DescriptorSetLayout::new(
99                "TextureAtlas Descriptor Set layout",
100                gfx.vulkan.device.clone(),
101                &create_info,
102            )
103            .context("Unable to create TextureAtlas descriptor set layout!")?
104        };
105        let descriptor_pool = {
106            let pool_sizes = [
107                vk::DescriptorPoolSize {
108                    ty: vk::DescriptorType::SAMPLER,
109                    descriptor_count: 1,
110                },
111                vk::DescriptorPoolSize {
112                    ty: vk::DescriptorType::SAMPLED_IMAGE,
113                    descriptor_count: MAX_TEXTURES,
114                },
115            ];
116            raii::DescriptorPool::new(
117                "TextureAtlas DescriptorPool",
118                gfx.vulkan.device.clone(),
119                &vk::DescriptorPoolCreateInfo {
120                    flags: vk::DescriptorPoolCreateFlags::UPDATE_AFTER_BIND,
121                    max_sets: 1,
122                    pool_size_count: pool_sizes.len() as u32,
123                    p_pool_sizes: pool_sizes.as_ptr(),
124                    ..Default::default()
125                },
126            )
127            .context("Error creating TextureAtlas descriptor pool!")?
128        };
129        let descriptor_set = {
130            let counts = [MAX_TEXTURES];
131            let mut descriptor_set_variable_descriptor_count_allocate_info =
132                vk::DescriptorSetVariableDescriptorCountAllocateInfo {
133                    descriptor_set_count: 1,
134                    p_descriptor_counts: counts.as_ptr(),
135                    ..Default::default()
136                };
137            let allocate_info = vk::DescriptorSetAllocateInfo {
138                descriptor_pool: descriptor_pool.raw,
139                descriptor_set_count: 1,
140                p_set_layouts: &descriptor_set_layout.raw,
141                ..Default::default()
142            }
143            .push_next(
144                &mut descriptor_set_variable_descriptor_count_allocate_info,
145            );
146            unsafe {
147                gfx.vulkan
148                    .allocate_descriptor_sets(&allocate_info)
149                    .context(
150                        "Unable to allocate TextureAtlas descriptor set!",
151                    )?[0]
152            }
153        };
154
155        let pipeline_layout = {
156            let ranges = [vk::PushConstantRange {
157                stage_flags: vk::ShaderStageFlags::VERTEX,
158                offset: 0,
159                size: 8 + 8 + 4,
160            }];
161            raii::PipelineLayout::new(
162                "TextureAtlas Layout",
163                gfx.vulkan.device.clone(),
164                &vk::PipelineLayoutCreateInfo {
165                    set_layout_count: 1,
166                    p_set_layouts: &descriptor_set_layout.raw,
167                    push_constant_range_count: ranges.len() as u32,
168                    p_push_constant_ranges: ranges.as_ptr(),
169                    ..Default::default()
170                },
171            )
172            .context("Unable to create effective pipeline layout")?
173        };
174
175        Ok(Self {
176            textures: vec![],
177            _sampler: sampler,
178            descriptor_set_layout,
179            _descriptor_pool: descriptor_pool,
180            descriptor_set,
181            pipeline_layout,
182        })
183    }
184
185    pub fn descriptor_set_layout(&self) -> &raii::DescriptorSetLayout {
186        &self.descriptor_set_layout
187    }
188
189    /// Binds the atlas's descriptor set for the frame.
190    ///
191    /// The atlas always binds to descriptor 0 as it should be compatible with
192    /// every other pipeline layout used during the frame and should never
193    /// need to be rebound.
194    pub fn bind_atlas_descriptor(&self, gfx: &Gfx, frame: &Frame) {
195        unsafe {
196            gfx.vulkan.cmd_bind_descriptor_sets(
197                frame.command_buffer(),
198                vk::PipelineBindPoint::GRAPHICS,
199                self.pipeline_layout.raw,
200                0,
201                &[self.descriptor_set],
202                &[],
203            );
204        }
205    }
206
207    /// Adds a texture to the atlas and returns the index which can be used in
208    /// vertices to reference the texture again.
209    pub fn add_texture(&mut self, gfx: &Gfx, texture: Texture) -> i32 {
210        let texture_index = self.textures.len() as i32;
211        let view = texture.view().raw;
212        self.textures.push(texture);
213
214        // Safe because this only writes to the newest texture index, which was
215        // never in use until after this method returns it.
216        unsafe {
217            let image_info = vk::DescriptorImageInfo {
218                sampler: vk::Sampler::null(),
219                image_view: view,
220                image_layout: vk::ImageLayout::SHADER_READ_ONLY_OPTIMAL,
221            };
222            gfx.vulkan.update_descriptor_sets(
223                &[vk::WriteDescriptorSet {
224                    dst_set: self.descriptor_set,
225                    dst_binding: 1,
226                    dst_array_element: texture_index as u32,
227                    descriptor_count: 1,
228                    descriptor_type: vk::DescriptorType::SAMPLED_IMAGE,
229                    p_image_info: &image_info,
230                    p_buffer_info: std::ptr::null(),
231                    p_texel_buffer_view: std::ptr::null(),
232                    ..Default::default()
233                }],
234                &[],
235            );
236        }
237
238        texture_index
239    }
